				<Script Src="~/scripts/niftycube.js" />

			</Scripts>
			<Styles>
				<Style href="~/scripts/niftyCorners.css" />
				<Style href="~/scripts/XFNHighlighter.css" />
				<Style href="~/scripts/lightbox.css" />
				<Style href="print.css" media="print" />
				<Style href="~/skins/_System/csharp.css" />
				<Style href="~/Admin/Resources/Scripts/helptip.css" />
			</Styles>
		</SkinTemplate>

3- Select skin named "curvy" from your admin panel
4- Enjoy
