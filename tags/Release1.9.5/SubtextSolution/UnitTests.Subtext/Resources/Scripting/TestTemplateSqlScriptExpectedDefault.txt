( 
	BlogId
)

WHERE PostConfig & 16 = 16 AND PostConfig & 1 = 1
GO

( EntryID, BlogId, WebCount, AggCount, WebLastUpdated, AggLastUpdated )

        EntryID, BlogId, WebCount, AggCount, WebLastUpdated, AggLastUpdated
    FROM [DotTextData].[dbo].[blog_EntryViewCount]
